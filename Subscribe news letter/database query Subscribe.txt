DROP DATABASE IF EXISTS SubscribeDB;

CREATE DATABASE SubscribeDB;

CREATE or REPLACE TABLE SubscribeDB.Subscribe

(

  Name varchar(40) NOT NULL default '0',

  Email   varchar(40) NOT NULL default '0',

  Address  varchar(50) NOT NULL default '0',

  Company   varchar(30) NOT NULL default '0',

  Profession   varchar(30) NOT NULL default '0',

  WorkPosition     varchar(30) NOT NULL default '0',

  PRIMARY KEY  (Name)

)
